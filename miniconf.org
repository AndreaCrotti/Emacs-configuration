** Add eventually
#+begin_src elisp :tangle yes
(defun org-add-eventually()
  "Adding a file to org-agenda when saved"
  (interactive)
  (if (string= major-mode "org-mode")
      (org-agenda-file-to-front)))

(add-hook 'before-save-hook 'org-add-eventually)
#+end_src
